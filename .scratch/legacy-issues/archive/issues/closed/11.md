     │ type RoleConfiguration struct {                                                                           │
     │     // Existing fields...                                                                                 │

     │ }                                                                                                         │
     │                                                                                                           │
     │ type RoleTypeConfig struct {                                                                              │
     │     Count        int               `json:"count"`        // Desired number of this type                   │
     │     EnabledCards map[string]bool   `json:"enabledCards"` // Which cards are available                     │
     │ }                                                                                                         │
     │                                                                                                           │
     │ State Management Rules                                                                                    │
     │                                                                                                           │
     │ 1. When user increases count:                                                                             │
     │   - Simply update the count                                                                               │
     │   - If count > enabled cards, show warning: "⚠️ Need to enable more Assassin cards (3 needed, 2 enabled)" │
     │   - Do NOT auto-enable anything                                                                           │
     │ 2. When user decreases count:                                                                             │
     │   - Keep existing card selections unchanged                                                               │
     │   - The game will use fewer of the enabled cards                                                          │
     │ 3. When user disables a card:                                                                             │
     │   - Allow the action                                                                                      │
     │   - If this causes count > enabled cards, show warning                                                    │
     │   - Do NOT prevent the action or auto-enable others                                                       │
     │                                                                                                           │

     │                                                                                                           │
     │ func MigrateRoleConfig(old *RoleConfiguration) *RoleConfiguration {                                       │
     │     new := &RoleConfiguration{                                                                            │
     │         // Copy existing fields...                                                                        │
     │         RoleTypes: make(map[string]*RoleTypeConfig),                                                      │
     │     }                                                                                                     │
     │                                                                                                           │
     │     // Initialize all role types with ALL cards enabled by default                                        │
     │     for category := range GetAllRoleCategories() {                                                        │
     │         new.RoleTypes[category] = &RoleTypeConfig{                                                        │
     │             Count: 0,                                                                                     │
     │             EnabledCards: make(map[string]bool),                                                          │
     │         }                                                                                                 │
     │                                                                                                           │
     │         // Enable ALL cards in this category                                                              │
     │         for _, card := range GetCardsByCategory(category) {                                               │
     │             new.RoleTypes[category].EnabledCards[card.Name] = true                                        │
     │         }                                                                                                 │
     │     }                                                                                                     │
     │                                                                                                           │
     │     // Set counts based on old config                                                                     │
     │     for roleName, enabled := range old.EnabledRoles {                                                     │
     │         if enabled {                                                                                      │
     │             category := GetRoleCategory(roleName)                                                         │
     │             new.RoleTypes[category].Count += old.RoleCounts[roleName]                                     │
     │         }                                                                                                 │
     │     }                                                                                                     │
     │                                                                                                           │
     │     return new                                                                                            │
     │ }                                                                                                         │
     │                                                                                                           │
     │ Implementation Details                                                                                    │
     │                                                                                                           │
     │ 1. Backend Endpoints:                                                                                     │
     │   - POST /room/{code}/config/role-type-count - Update count for a role type                               │
     │   - POST /room/{code}/config/card-toggle - Toggle individual card                                         │
     │   - Keep existing preset endpoint                                                                         │
     │ 2. Frontend Behavior:                                                                                     │
     │   - Collapsible sections with Datastar                                                                    │
     │   - Real-time validation on any change                                                                    │
     │   - Clear visual indicators for state                                                                     │
     │   - No automatic selections                                                                               │
     │ 3. Validation Logic:                                                                                      │
     │ func (rt *RoleTypeConfig) Validate() (bool, string) {                                                     │
     │     enabledCount := 0                                                                                     │
     │     for _, enabled := range rt.EnabledCards {                                                             │
     │         if enabled {                                                                                      │
     │             enabledCount++                                                                                │
     │         }                                                                                                 │
     │     }                                                                                                     │
     │                                                                                                           │
     │     if rt.Count > enabledCount {                                                                          │
     │         return false, fmt.Sprintf("Need %d cards, but only %d enabled",                                   │
     │                                  rt.Count, enabledCount)                                                  │
     │     }                                                                                                     │
     │     return true, ""                                                                                       │
     │ }                                                                                                         │
